# Credit Card
The `dda-creditcard-field` component is a customizable button element that can be used in various UI scenarios

### Usage in React

```jsx
import React from 'react'
import { DdaCreditcardField } from "@dubai-design-system/components-react";


const App = () => {
  return (
    <div>
        <DdaCreditcardField
            placeholder="0000 - 0000 - 0000 - 0000"
            label="Card Number"
            restrict_input="false"
            value=""
            card_icon="https://www.mastercard.com/content/dam/public/brandcenter/assets/images/logos/mclogo-for-footer.svg"
            size="default"
            helper_text="Helper or footer text"
            error_message=""
            validation_type="normal"
            custom_class=""
            component_mode=""
            input_id="input"
            aria_label="credit-card-field"
            onchange="console.log('changed')"
        ></DdaCreditcardField>
    </div>
 )
}
```
