# Radio Button
The `dda-radiobutton` component is a customizable button element that can be used in various UI scenarios

### Usage in React

```jsx
import React from 'react'
import { DdaRadiobutton } from "@dubai-design-system/components-react";




const App = () => {
  return (
    <div>
        <DdaRadiobutton
            title_text="Radio Button Title"
            checked="false"
            size="md"
            variants="normal"
            supporting="Supporting Text"
            group_name="radiogroup"
            input_id="testId"
            custom_class=""
            component_mode=""
            aria_label="radio-button"
            onclick="console.log('clicked')"
        ></DdaRadiobutton>
    </div>
 )
}

```
