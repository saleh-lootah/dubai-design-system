# Sticky Footer
The `dda-sticky-footer` component is customizable that can be used in various UI scenarios

#### Usage in React

```jsx
import React from 'react'
import { DdaStickyFooter } from "@dubai-design-system/components-react";



const App = () => {
  return (
<DdaStickyFooter
      class="hydrated"

      /** Left Section Props */
      leftIconHref="#"
      leftIconSrc="https://via.placeholder.com/24"
      leftIconAlt="Left Icon 1"
      leftIconTooltip="Tooltip for Left Icon 1"

      leftIcon2Href="#"
      leftIcon2Src="https://via.placeholder.com/24"
      leftIcon2Alt="Left Icon 2"
      leftIcon2Tooltip="Tooltip for Left Icon 2"

      leftIcon3Href="#"
      leftIcon3Src="https://via.placeholder.com/24"
      leftIcon3Alt="Left Icon 3"
      leftIcon3Tooltip="Tooltip for Left Icon 3"
      leftIcon3Text="Services"

      /** Middle Section Props */
      middleLogo1Href="#"
      middleLogo1Src="https://via.placeholder.com/100"
      middleLogo1Alt="Middle Logo 1"
      middleLogo1Tooltip="Tooltip for Middle Logo 1"

      middleLogo2Href="#"
      middleLogo2Src="https://via.placeholder.com/100"
      middleLogo2Alt="Middle Logo 2"
      middleLogo2Tooltip="Tooltip for Middle Logo 2"

      middleLogo3Href="#"
      middleLogo3Src="https://via.placeholder.com/100"
      middleLogo3Alt="Middle Logo 3"
      middleLogo3Tooltip="Tooltip for Middle Logo 3"

      /** Right Section Props */
      rightButton1Href="#"
      rightButton1Text="Location"
      rightButton1Icon="https://via.placeholder.com/16"

      rightButton2Href="#"
      rightButton2Text="Support"
      rightButton2Icon="https://via.placeholder.com/16"

      rightButton3Href="#"
      rightButton3Text="Contact"
      rightButton3Icon="https://via.placeholder.com/16"

      rightIcon1Href="#"
      rightIcon1Src="https://via.placeholder.com/24"
      rightIcon1Alt="Right Icon 1"
      rightIcon1Tooltip="Tooltip for Right Icon 1"

      rightIcon2Href="#"
      rightIcon2Src="https://via.placeholder.com/24"
      rightIcon2Alt="Right Icon 2"
      rightIcon2Tooltip="Tooltip for Right Icon 2"
    />
 )
}
```
